#  TWO RANDOM NUMBERS
      CALL RMMAR( RD,2,1 )
#  GENERATE <P_T>   (REFERENCE??)
      B   = ZN * (ZN - 1.D0)
      ZZ  = SQRT(1.D0/RD(1) - 1.D0)

      PTY = XPT * FACT * SIN(Z)
