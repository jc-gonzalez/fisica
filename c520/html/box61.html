#-----------------------------------------------------------------------

#KEEP,NCOUNT.

     *                 (CURPAR(4), PHI ),  (CURPAR(5), H    ),
     *                 (CURPAR(6), T   ),  (CURPAR(7), X    ),
     *                 (CURPAR(8), Y   ),  (CURPAR(9), CHI  ),
     *                 (CURPAR(10),BETA),  (CURPAR(11),GCM  ),

#KEEP,RANDPA.

      DOUBLE PRECISION FAC,U1,U2

#KEEP,RUNPAR.
      COMMON /RUNPAR/  FIXHEI,THICK0,HILOECM,HILOELB,
     *                 STEPFC,NRRUN,NSHOW,PATAPE,MONIIN,

      IF ( DEBUG ) WRITE(MDEBUG,444) (CURPAR(I),I=1,9)
  444 FORMAT(' BOX61 : CURPAR=',1P,9E10.3)
 
#  ANNIHILATION
      IF ( ITYPE .EQ. 15  .OR.  ITYPE .EQ. 25 ) THEN
        NCOUN(3) = NCOUN(3) + 1
        CALL RMMAR( RD,1,1 )
        IF ( RD(1)*SIGMA .LE. SIGANN ) THEN
          NCOUN(4) = NCOUN(4) + 1
          CALL NIHILA
          RETURN
        ENDIF
      ENDIF
 
#  INTERACTION (ISOBAR EXCITATION)
#  SELECT LIGHT ISOBAR AND NUCLEON MASSES
      CA = C(3)
      CB = PAMA(14)
 
#  DECIDE WHETHER ISOBAR GOES FORWARD OR BACKWARD
#  FORWARD IF KIND = 0, BACKWORD IF KIND = 1
      CALL RMMAR( RD,1,1 )
      IF ( RD(1) .LE. 0.5 ) THEN
        KIND = 0
      ELSE
        KIND = 1
      ENDIF
 
#  LIGHT ISOBAR
      CALL ISOBAR( ECM,KIND,CA,CB,1 )
 
#  NUCLEON
      CALL SINGLE( ECM,1-KIND,CB,CA )
 
      RETURN
