      SUBROUTINE GRANOR(A,B)
 
#-----------------------------------------------------------------------
#  TWO GAUSSIAN DISTRIBUTED RANDOM NUMBERS (MEAN=0, SIGMA=1)
#
#  RANDOM NUMBER GENERATORS USED BY THE GHEISHA ROUTINES
#  ADAPTED FOR USE WITH THE CORSIKA RANDOM NUMBER GENERATORS
#
#  DESIGN  : J. KNAPP   IK1  FZK KARLSRUHE
#-----------------------------------------------------------------------
 
      REAL RD(2)
#-----------------------------------------------------------------------
 
      CALL RMMAR(RD,2,1)
      U1 = SQRT(-2.*LOG(RD(1)))
      U2 = RD(2) * 6.28318530718
      A = COS(U2) * U1
      B = SIN(U2) * U1
      RETURN
      END
